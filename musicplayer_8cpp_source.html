46 void MusicPlayer::start()
47 {
48  while(true)
49  {
50  if(hasError())
51  break;
52 
53  if(_waiting_next_song)
54  handlePlayRequest();
55 
56  auto request = getRequest();
57  switch(request)
58  {
59  case ControlRequest::Play:
60  handlePlayRequest();
61  break;
62  case ControlRequest::Stop:
63  handleStopRequest();
64  break;
65  case ControlRequest::Pause:
66  _waiting_next_song = false;
67  if(_handles_opened)
68  setIsPlaying(false);
69  break;
70  case ControlRequest::Resume:
71  if(_handles_opened)
72  setIsPlaying(true);
73  break;
74  case ControlRequest::Next:
75  handleNextRequest();
76  break;
77  case ControlRequest::Prev:
78  handlePrevRequest();
79  break;
80  case ControlRequest::VolumeUp:
81  {
82  double base, really, rva;
83  mpg123_getvolume(_mpg_handle, &base, &really, &rva);
84  if(really + VOLUME_PRECISION > MAX_VOLUME)
85  break;
86  mpg123_volume_change(_mpg_handle, VOLUME_PRECISION);
87  break;
88  }
89  case ControlRequest::VolumeDown:
90  mpg123_volume_change(_mpg_handle, -VOLUME_PRECISION);
91  break;
92  default:
93  break;
94  }
95 
96  if(!isPlaying())
97  {
98  std::this_thread::sleep_for(std::chrono::seconds(1));
99  continue;
100  }
101 
102  size_t written_bytes_count;
103  int err = mpg123_read(_mpg_handle, _out_block, _out_block_size, &written_bytes_count);
104  out123_play(_out_handle, _out_block, written_bytes_count);
105  if(written_bytes_count == 0 || err != MPG123_OK)
106  handleNextRequest();
107  }
